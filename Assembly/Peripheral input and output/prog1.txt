LIGHTCHARSQ	FCC	'Qq'

LIGHTCHARSE	FCC	'Ee'

NLCHARSC	EQU	2

DTWO	FCB	2
DTHREE	FCB	2
TEMP	FCB	2

		ORG	$9300
	LDAA	#00
	STAA	$1000
	STAA	DTHREE
	STAA	DTWO
MAINLOOP
		JSR	INCHAR
		LDX	#LIGHTCHARSQ
		LDAB	#NLCHARSQ
		JSR	LOOKUP
		BCS	QFOUND
		LDX	#LIGHTCHARSZ
		LDAB	#NLCHARSZ
		JSR	LOOKUP
		BCS	ZFOUND
		LDX	#LIGHTCHARSE
		LDAB	#NLCHARSE
		JSR	LOOKUP
		BCS	EFOUND
		LDX	#LIGHTCHARSC
		LDAB	#NLCHARSC
		JSR	LOOKUP
		BCS	CFOUND
		BRA	MAINLOOP
QFOUND
		LDAA	#%10
		STAA	DTHREE
		BRA	MOTHERBRAIN
ZFOUND
		LDAA	#%00
		STAA	DTHREE
		BRA	MOTHERBRAIN
EFOUND
		LDAA	#%01
		STAA	DTWO
		BRA	MOTHERBRAIN
CFOUND
		LDAA	#%00
		STAA	DTWO
		BRA	MOTHERBRAIN

MOTHERBRAIN	
		CLC
		LDAA	DTHREE
		ORAA	DTWO
		BEQ	OFF
		CMPA	#01
		BEQ	RIGHTON
		CMPA	#02
		BEQ	MIDON
		CMPA	#03
		BEQ	BOTHON
	

OFF
		LDAA	#$00
		STAA	$1000
		BRA	MAINLOOP
RIGHTON
		LDAA	#$10
		STAA	$1000
		BRA	MAINLOOP
MIDON
		LDAA	#$20
		STAA	$1000
		BRA	MAINLOOP
BOTHON
		LDAA	#$30
		STAA	$1000
		BRA	MAINLOOP
