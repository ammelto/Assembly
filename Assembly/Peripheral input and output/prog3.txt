TFLG2	EQU 	$1025
PORTA	EQU	$1000

; Various defines go here ...
	ORG $9000
INFO 	FCC 'Your name'
	FCB 	10
	FCC 	'ENSC 3213'
	FCB 	10
	FCC 	'Date the program was last changed'
	FCB	 10,10,4

	ORG $9100
	LDX	 #INFO
	JSR 	OUTSTRG

; CLEAR THE FLAG TO INITIALIZE
LOOP1
	LDAA 	#%10000000
	STAA	 TFLG2

;WAIT FOR THE FLAG TO BE SET
LOOP2
	LDAA	 TFLG2
	ANDA	 #%10000000
	BEQ	 LOOP2

;NOW TOGGLE PA4
	LDAA	 #%00010000
	EORA	 PORTA
	STAA 	 PORTA

;DO IT ALL OVER AGAIN
	BRA 	LOOP1
