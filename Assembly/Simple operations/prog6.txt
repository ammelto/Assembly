OUTLHLF EQU	$FFB2
OUTRHLF EQU	$FFB5
INCHAR	EQU 	$FFCD
OUTSTRG	EQU	$FFC7
OUTCRLF EQU	$FFC4

	ORG	$8000
MAIN
	LDAA	#$0
	STAA	VALUE
	LDAA	#$1
	STAA	DELTA
	LDAA	#$B
	STAA	COUNT
LOOP	BEQ	END
	LDAA	VALUE
	JSR	OUTLHLF
	LDAA	VALUE
	JSR	OUTRHLF
	LDX	#PROMPT
	JSR	OUTSTRG
	LDAA	VALUE
	ADDA	DELTA
	STAA	VALUE
	LDAA	DELTA
	ADDA	#$2
	STAA	DELTA
	DEC	COUNT
	BRA	LOOP
END	SWI



	ORG	$9000

PROMPT	FCC	''
	FCB	4
DELTA		FCB	1
VALUE		FCB	1
COUNT		FCB	1
